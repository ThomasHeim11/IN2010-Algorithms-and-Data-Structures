ALGORITHM: Balansert BST med bruka av Heap
Input: En heap (PriorityQueue) med tall
Output: Tallene skrevet ut i balansert rekkef√∏lge

Procedure printBalancedOrder(heap)
    tallListe <- ny liste av heltall
    while heap ikke er tom do
        tallListe.leggTil(heap.poll())
    
    printBalancedOrderHelper(tallListe, 0, |tallListe| -1)

Procedure printBalancedOrderHelper(tallListe,start,end)
    if start > end then
        return
    
    mid <- (start + end)/ 2
    PRINT(tallListe.get(mid))

    printBalancedOrderHelper(tallListe,start,end - 1)
    printBalancedOrderHelper(tallListe,mid + 1,end)
